 host = "http://localhost:8081";
    readUser(){
        const doc = document.getElementById("doc").value;
        fetch(`${host}/users/consultar/${doc}`)
        .then(res => res.json())
        .then(res => {
            document.getElementById("typeDoc").value=res.typeDoc;
            document.getElementById("nom").value=res.nom;
            document.getElementById("mail").value=res.mail;
            document.getElementById("tel").value=res.tel;
            document.getElementById("dir").value=res.dir;
            document.getElementById("datetime").value=res.datetime;
        });
    }
    
    addUser(){
        const typeDoc = document.getElementById("typeDoc").value;
        const doc = document.getElementById("doc").value;
        const nom = document.getElementById("nom").value;
        const mail = document.getElementById("mail").value;
        const pass = document.getElementById("pass").value;
        const tel = document.getElementById("tel").value;
        const dir = document.getElementById("dir").value;
        const datetime = document.getElementById("datetime").value;
        fetch(`${host}/guardar`,{
            headers: {contentType: "application/json"},
            method: "POST",
            body: JSON.stringify({typeDoc, doc, nom, mail, pass, tel, dir, datetime})           
        })
        .then(dato => dato.json())
        .then(dato => alert(dato.msg))
        .catch(error => alert(error));
    };
